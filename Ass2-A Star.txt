import heapq
# heapq module is used to implement Priority Queue (Min Heap)
# The node with smallest f(n) value is selected first


def astar(maze, start, end):

    # Get total rows and columns of maze
    rows, cols = len(maze), len(maze[0])

    # Priority Queue for nodes to explore
    open_list = []

    # Insert starting node with priority 0
    heapq.heappush(open_list, (0, start))

    # Dictionary to store previous node path
    came_from = {}

    # Cost from start node to current node
    g_score = {start: 0}

    # Continue until queue becomes empty
    while open_list:

        # Remove node having smallest f(n)
        _, current = heapq.heappop(open_list)

        # If destination reached
        if current == end:

            # Store final path
            path = []

            # Backtrack path from destination to source
            while current in came_from:
                path.append(current)
                current = came_from[current]

            # Add start node
            path.append(start)

            # Reverse path because backtracking gives reverse order
            return path[::-1]

        # Explore neighbors in 4 directions
        # Up, Down, Left, Right
        for dx, dy in [(-1,0), (1,0), (0,-1), (0,1)]:

            # Calculate neighbor coordinates
            neighbor = (current[0] + dx, current[1] + dy)

            # Check:
            # 1. Neighbor inside maze boundary
            # 2. Neighbor is not wall (1)
            if (0 <= neighbor[0] < rows and
                0 <= neighbor[1] < cols and
                maze[neighbor[0]][neighbor[1]] != 1):

                # Calculate temporary cost
                temp_g = g_score[current] + 1

                # If node not visited OR shorter path found
                if neighbor not in g_score or temp_g < g_score[neighbor]:

                    # Update shortest distance
                    g_score[neighbor] = temp_g

                    # Heuristic Function
                    # Manhattan Distance
                    h = abs(neighbor[0] - end[0]) + abs(neighbor[1] - end[1])

                    # f(n) = g(n) + h(n)
                    f = temp_g + h

                    # Insert neighbor into priority queue
                    heapq.heappush(open_list, (f, neighbor))

                    # Store parent node
                    came_from[neighbor] = current

    # If path does not exist
    return None


# Function to print maze and final path
def print_maze(maze, path):

    for r in range(len(maze)):

        for c in range(len(maze[0])):

            # Print '.' for path
            if (r, c) in path:
                print('.', end=' ')

            # Print 'X' for wall
            elif maze[r][c] == 1:
                print('X', end=' ')

            # Print '-' for empty cell
            else:
                print('-', end=' ')

        print()


# ---------------- INPUT SECTION ----------------

# Input number of rows
rows = int(input("Enter rows: "))

# Input number of columns
cols = int(input("Enter cols: "))

# Empty maze list
maze = []

print("Enter maze values:")

# Take maze input row by row
for i in range(rows):

    # Example: 0 1 0 0
    row = list(map(int, input().split()))

    # Add row into maze
    maze.append(row)

# Input starting coordinate
# Example: 0 0
start = tuple(map(int, input("Enter start point: ").split()))

# Input ending coordinate
# Example: 4 4
end = tuple(map(int, input("Enter end point: ").split()))


# ---------------- FUNCTION CALL ----------------

# Call A* Algorithm
path = astar(maze, start, end)


# ---------------- OUTPUT SECTION ----------------

# If path exists
if path:

    print("\nPath Found:")

    # Print shortest path coordinates
    print(path)

    print("\nMaze Path:")

    # Print maze with final path
    print_maze(maze, path)

# If no path found
else:
    print("No Path Found")
